void setup() {
  // Konfiguriere Kommunikation zum PC
  Serial.begin(9600);
}

void loop() {
  // sende Text zum PC
  Serial.println("Hallo!");
  // warte 1000ms, also eine Sekunde
  delay(1000);
}

Was haben wir in Blink gelernt?


int led = 13; // Variable für Pin anlegen
						
void setup() {
  Serial.begin(9600);
  pinMode(led, OUTPUT);  
}

void loop() {
  Serial.println("LED an");
  digitalWrite(led, HIGH); // schalte LED an
  delay(1000);
  
  Serial.println("LED aus");
  digitalWrite(led, LOW); // schalte LED aus
  delay(1000);
}
